Handlebars Extended

Handlebars is a simple templating language.
It uses a template and an input object to generate HTML or other text formats.
Here, I would like to explain how to use the functions added to extend Handlebars.
See here for the basic usage of Handlebars. 

npm i handlebars-extd

Currently v1.0.4 • Install • Usage • Comparison • Date • HTML • Math • String • GitHub repo • Changelog

Install

Install with npm:

npm install --save handlebars-extd

Usage

Importing"handlebars-extd" will return an extended "Handlebars" with built-in helpers and more.

import hbs from 'handlebars-extd';

  • opr

    Compare two values using operators.

    Params:
    • any val1 First value to be compared with second.
    • '=='|'==='|'!='|'!=='|'<'|'>'|'<='|'>=' opr The operator to use. Operators must be enclosed in quotes: `">"`, `"="`, `"<="`, and so on.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{opr a '===' b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (opr a '===' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '==' b}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{#if (opr a '==' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '!==' b}}")({"a":4,"b":3});

// Output: true
hbs.compile("{{#if (opr a '!==' b)}}true{{else}}false{{/if}}")({"a":4,"b":3});

// Output: false
hbs.compile("{{opr a '!=' b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (opr a '!=' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '<' b}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{#if (opr a '<' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{opr a '>' b}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{#if (opr a '>' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{opr a '<=' b}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{#if (opr a '<=' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{opr a '>=' b}}")({"a":3,"b":3});

// Output: true
hbs.compile("{{#if (opr a '>=' b)}}true{{else}}false{{/if}}")({"a":3,"b":3});
  • eq

    Determine whether or not two values are equal (===).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{eq a b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (eq a b)}}true{{else}}false{{/if}}")({"a":"3","b":3});
  • eqw

    Determine whether or not two values are equal (==) i.e weak checking.

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: true
hbs.compile("{{eqw a b}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{#if (eqw a b)}}true{{else}}false{{/if}}")({"a":"3","b":3});
  • neq

    Determine whether or not two values are not equal (!==).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: true
hbs.compile("{{neq a b}}")({"a":4,"b":3});

// Output: true
hbs.compile("{{#if (neq a b)}}true{{else}}false{{/if}}")({"a":4,"b":3});
  • neqw

    Determine whether or not two values are not equal (!=) weak checking.

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{neqw a b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (neqw a b)}}true{{else}}false{{/if}}")({"a":"3","b":3});
  • lt

    Check for less than condition (a < b).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: true
hbs.compile("{{lt a b}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{#if (lt a b)}}true{{else}}false{{/if}}")({"a":2,"b":3});
  • gt

    Check for greater than condition (a > b).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{gt a b}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{#if (gt a b)}}true{{else}}false{{/if}}")({"a":2,"b":3});
  • lte

    Check for less than or equals condition (a <= b).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{lte a '<=' b}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{#if (lte a '<=' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});
  • gte

    Check for greater than or equals condition (a >= b).

    Params:
    • any val1 First value to be compared with second.
    • any val2 Second value to be compared with first.
    Examples
    // Output: false
hbs.compile("{{gte a '>=' b}}")({"a":3,"b":3});

// Output: false
hbs.compile("{{#if (gte a '>=' b)}}true{{else}}false{{/if}}")({"a":3,"b":3});
  • ifx

    Helper to imitate the ternary '?:' conditional operator.

    Params:
    • any val1 Value to return when the condition holds true.
    • any val2 Value to return when the condition is false (Optional).
    Examples
    // Output: foo
hbs.compile("{{ifx true a b}}")({"a":"foo","b":"bar"});

// Output: bar
hbs.compile("{{ifx false a b}}")({"a":"foo","b":"bar"});
  • not

    Logical NOT of any expression.

    Params:
    • any expr Any expression.
    Examples
    // Output: false
hbs.compile("{{not true}}")();

// Output: false
hbs.compile("{{#if (not true)}}true{{else}}false{{/if}}")();

// Output: true
hbs.compile("{{not false}}")();

// Output: true
hbs.compile("{{#if (not false)}}true{{else}}false{{/if}}")();
  • empty

    Check if it is empty.
    If the value is an array, returns true if there are no elements.
    If the value is a string, the leading and trailing spaces are trimmed and then checked.

    Params:
    • any val Character strings, arrays, objects, etc. to be checked.
    Examples
    // If the value is an array.
// Output: true
hbs.compile("{{empty val}}")({"val":[]});

// Output: true
hbs.compile("{{#if (empty val)}}true{{else}}false{{/if}}")({"val":[]});

// Output: false
hbs.compile("{{empty val}}")({"val":["foo"]});

// If the value is a string.
// Output: false
hbs.compile("{{empty val}}")({"val":"Hello"});

// Output: true
hbs.compile("{{empty val}}")({"val":""});

// Output: true
hbs.compile("{{empty val}}")({"val":" "});
  • notEmpty

    Check that it is not empty.
    If the value is an array, returns true if there is an element.
    If the value is a string, the leading and trailing spaces are trimmed and then checked.

    Params:
    • any val Character strings, arrays, objects, etc. to be checked.
    Examples
    // If the value is an array.
// Output: false
hbs.compile("{{empty val}}")({"val":[]});

// Output: false
hbs.compile("{{#if (empty val)}}true{{else}}false{{/if}}")({"val":[]});

// Output: true
hbs.compile("{{empty val}}")({"val":["foo"]});

// If the value is a string.
// Output: true
hbs.compile("{{empty val}}")({"val":"Hello"});

// Output: false
hbs.compile("{{empty val}}")({"val":""});

// Output: false
hbs.compile("{{empty val}}")({"val":" "});
  • count

    Determine the length of an array.

    Params:
    • any[] coll Array whose elements to be counted.
    Examples
    // Output: 2
hbs.compile("{{count coll}}")({"coll":["foo","bar"]});
  • and

    Returns the boolean AND of two or more parameters passed i.e it is true iff all the parameters are true.

    Params:
    • ...any args Any number of boolean parameters.
    Examples
    // Output: true
hbs.compile("{{and a b}}")({"a":true,"b":true});

// Output: true
hbs.compile("{{#if (and a b)}}true{{else}}false{{/if}}")({"a":true,"b":true});

// Output: false
hbs.compile("{{and a b}}")({"a":false,"b":true});

// Output: false
hbs.compile("{{#if (and a b)}}true{{else}}false{{/if}}")({"a":false,"b":true});
  • or

    Returns the boolean OR of two or more parameters passed i.e it is true if any of the parameters is true.

    Params:
    • ...any args Any number of boolean parameters.
    Examples
    // Output: true
hbs.compile("{{or a b}}")({"a":true,"b":false});

// Output: true
hbs.compile("{{#if (or a b)}}true{{else}}false{{/if}}")({"a":true,"b":false});

// Output: false
hbs.compile("{{or a b}}")({"a":false,"b":false});

// Output: false
hbs.compile("{{#if (or a b)}}true{{else}}false{{/if}}")({"a":false,"b":false});
  • coalesce

    Returns the first non-falsy value from the parameter list.
    Checks for the first non-false parameter.

    Params:
    • ...any args Any number of parameters.
    Examples
    // Output: foo
hbs.compile("{{coalesce a b c}}")({"a":"foo","b":"bar","c":"baz"});

// Output: bar
hbs.compile("{{coalesce a b c}}")({"a":"","b":"bar","c":"baz"});
  • includes

    Returns true if the array contains a particular value, false if it does not.

    Params:
    • any[] coll The array.
    • any val The value to search for.
    • boolean strict FALSE for non-strict checking. TRUE by default.
    Examples
    // Output: true
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":2});

// Output: true
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":2});

// Output: false
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":10});

// Output: false
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":10});

// Output: false
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":"3"});

// Output: false
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":"3"});

// Output: true
hbs.compile("{{includes coll a strict}}")({"coll":[1,2,3,4],"a":"3","strict":false});

// Output: true
hbs.compile("{{#if (includes coll a strict)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":"3","strict":false});

Date

Summary

Helper Description
formatDate Use moment to format the date.
  • formatDate

    Use moment to format the date.

    Params:
    • string format A format string based on moment.
    • string date Date string to format.
    • string|string[] locale Language or language-country locale string (or array of strings) available in https://github.com/moment/moment/tree/develop/locale.
    Examples
    // Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date}}")({"date":"2021-10-24T23:11:14.275Z"});

// Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date 'jp'}}")({"date":"2021-10-24T23:11:14.275Z"});

// Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date 'es'}}")({"date":"2021-10-24T23:11:14.275Z"});

HTML

Summary

Helper Description
classIf Set the CSS classes if the condition is true.
selectedIf Sets the selected attribute if the condition is true.
checkedIf Sets the checked attribute if the condition is true.
options Generates a select drop-down option list.
  • classIf

    Set the CSS classes if the condition is true.

    Params:
    • boolean expr Condition to be checked.
    • string cls CSS class to set if the condition is true.
    Examples
    // Output: foo
hbs.compile("{{classIf expr 'foo'}}")({"expr":true});
  • selectedIf

    Sets the selected attribute if the condition is true.

    Params:
    • boolean expr Condition to be checked.
    Examples
    // Output: selected
hbs.compile("{{selectedIf expr}}")({"expr":true});
  • checkedIf

    Sets the checked attribute if the condition is true.

    Params:
    • boolean expr Condition to be checked.
    Examples
    // Output: checked
hbs.compile("{{checkedIf expr}}")({"expr":true});
  • options

    Generates a select drop-down option list.

    Params:
    • {[key: string]: string}[] data List of data.
    • {value: string, text: string} opts Key names for the selected and displayed values of option in the data element (Optional).
    Examples
    // Output: <option value="1">foo</option>
//         <option value="2" selected>bar</option>
hbs.compile("{{{options data selected='2'}}}")({"data":[{"value":1,"text":"foo"},{"value":2,"text":"bar"}]});

// Output: <option value="392" selected>JAPAN</option>
//         <option value="840">UNITED STATES</option>
hbs.compile("{{{options data selected='392' value='code' text='name'}}}")({"data":[{"code":392,"name":"JAPAN"},{"code":840,"name":"UNITED STATES"}]});

Math

Summary

Helper Description
add Calculates the sum of two numbers.
sub Calculates the difference of the given values.
ceil Round up the value.

  • add

    Calculates the sum of two numbers.

    Params:
    • number|string val1 The first number.
    • number|string val2 The second number.
    Examples
    // Output: 3
hbs.compile("{{add a b}}")({"a":1,"b":2});
  • sub

    Calculates the difference of the given values.

    Params:
    • number|string val1 The first number.
    • number|string val2 The second number.
    Examples
    // Output: 3
hbs.compile("{{sub a b}}")({"a":5,"b":2});

    Round up the value.

    Params:
    • number|string val Number to be rounded to nearest greater integer.
    Examples
    // Output: 6
hbs.compile("{{ceil a}}")({"a":5.6});
  • floor

    Rounds down a number.

    Params:
    • number|string val Number to be rounded to nearest lower integer.
    Examples
    // Output: 5
hbs.compile("{{floor a}}")({"a":5.6});
  • abs

    Returns an absolute value.

    Params:
    • number|string val Number to perform absolute value operation on.
    Examples
    // Output: 5.6
hbs.compile("{{abs a b}}")({"a":-5.6});

String

Summary

Helper Description
slice Extract a portion of a string.
nltobr Replace \n with <br> tags.
sprintf Returns a string produced according to the formatting string format.
It uses sprintf-js internally.
Check https://github.com/alexei/sprintf.js for more information.
lowercase Changes the string to lowercase.
uppercase Changes the string to uppercase.
concat Concat two or more strings.
join Join the elements of an array using a delimeter.
split Split string by the given character.
  • slice

    Extract a portion of a string.

    Params:
    • string val Target string.
    • number beginIdx The zero-based index at which to begin extraction.
    • number endIdx The zero-based index before which to end extraction. The character at this index will not be included.
    Examples
    // Output: Just
hbs.compile("{{slice str 0 4}}")({"str":"Just Wow"});

// Output: Wow
hbs.compile("{{slice str 5}}")({"str":"Just Wow"});

// Output:  Wow
hbs.compile("{{slice str -4}}")({"str":"Just Wow"});
  • nltobr

    Replace \n with <br> tags.

    Params:
    • string val Target string.
    Examples
    // Output: It's<br>just<br>now
hbs.compile("{{{nltobr str}}}")({"str":"It's\njust\nnow"});
  • sprintf

    Returns a string produced according to the formatting string format.
    It uses sprintf-js internally.
    Check https://github.com/alexei/sprintf.js for more information.

    Params:
    • string format Format string.
    • ...any args Any number of parameters/values.
    Examples
    // Argument swapping:
// Output: Hello Dolly!
hbs.compile("{{sprintf '%s %s!' str1 str2}}")({"str1":"Hello","str2":"Dolly"});

// Output: foo bar 55 baz 20
hbs.compile("{{sprintf '%s %s %d %s %d' 'foo' 'bar' 55 'baz' '20'}}")();

// Named arguments:
// Output: Hello Dolly
hbs.compile("{{sprintf 'Hello %(name)s' user}}")({"user":{"name":"Dolly"}});

// Output: Hello Dolly
hbs.compile("{{sprintf 'Hello %(name)s' name=str}}")({"str":"Dolly"});
  • lowercase

    Changes the string to lowercase.

    Params:
    • string val Target string.
    Examples
    // Output: just wow
hbs.compile("{{lowercase str}}")({"str":"JUST WOW"});
  • uppercase

    Changes the string to uppercase.

    Params:
    • string val Target string.
    Examples
    // Output: JUST WOW
hbs.compile("{{uppercase str}}")({"str":"just wow"});
  • concat

    Concat two or more strings.

    Params:
    • ...string args Any number of string arguments.
    Examples
    // Output: Hello world!
hbs.compile("{{concat 'Hello' ' world' '!'}}")();
  • join

    Join the elements of an array using a delimeter.

    Params:
    • string[] coll An array of elements to be joined.
    • string delim A delimiter that joins array elements.
    Examples
    // Output: Hands & legs & feet
hbs.compile("{{{join coll ' & '}}}")({"coll":["Hands","legs","feet"]});
  • split

    Split string by the given character.

    Params:
    • string value The value to split for.
    • string separator A character that delimits the substrings in this string. Default is a comma.
    Examples
    // Output: ['a', 'b', 'c']
hbs.compile("{{split list}}")({"list":"a,b,c"});

// Output: <ul>
//           <li>a</li>
//           <li>b</li>
//           <li>c</li>
//         </ul>
hbs.compile(`<ul>
              {{#each (split list ',')}}
                <li>{{this}}</li>
              {{/each}}
            </ul>`)({"list":"a,b,c"});
